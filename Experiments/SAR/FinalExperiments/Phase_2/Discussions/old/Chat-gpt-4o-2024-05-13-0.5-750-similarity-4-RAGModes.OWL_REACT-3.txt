### Example of Answering Competency Questions
### CQ1: What are the most important weather variables that can cause forest fire?
:WeatherCondition rdf:type owl:Class ;
    rdfs:subClassOf [
        rdf:type owl:Class ;
        owl:oneOf ( :Temperature :WindSpeed :Humidity )
    ] .

### CQ2: What are the current measurements for these weather variables?
:WeatherMeasurement rdf:type owl:Class ;
    rdfs:subClassOf :WeatherCondition ;
    rdfs:label "WeatherMeasurement" ;
    rdfs:comment "Represents the current measurements of weather variables." .

:hasCurrentMeasurement rdf:type owl:DatatypeProperty ;
    rdfs:domain :WeatherMeasurement ;
    rdfs:range xsd:float ;
    rdfs:label "hasCurrentMeasurement" ;
    rdfs:comment "Specifies the current measurement of a weather variable." .

### CQ3: What is the forecast for the weather in this location?
:WeatherForecast rdf:type owl:Class ;
    rdfs:subClassOf :WeatherCondition ;
    rdfs:label "WeatherForecast" ;
    rdfs:comment "Represents the weather forecast for a specific location." .

:hasForecast rdf:type owl:DatatypeProperty ;
    rdfs:domain :WeatherForecast ;
    rdfs:range xsd:string ;
    rdfs:label "hasForecast" ;
    rdfs:comment "Specifies the weather forecast." .

### CQ4: Where did the incident take place?
:Incident rdf:type owl:Class ;
    rdfs:subClassOf [
        rdf:type owl:Class ;
        owl:intersectionOf ( :Incident [
            rdf:type owl:Restriction ;
            owl:onProperty :hasLocation ;
            owl:someValuesFrom xsd:string
        ] )
    ] .

### CQ5: What is the priority of an incident during a forest fire disaster?
:Incident rdf:type owl:Class ;
    rdfs:subClassOf [
        rdf:type owl:Class ;
        owl:intersectionOf ( :Incident [
            rdf:type owl:Restriction ;
            owl:onProperty :hasPriority ;
            owl:someValuesFrom xsd:string
        ] )
    ] .

### CQ6: What incidents during forest fires are the most urgent?
:UrgentIncident rdf:type owl:Class ;
    rdfs:subClassOf :Incident ;
    rdfs:label "UrgentIncident" ;
    rdfs:comment "Represents the most urgent incidents during forest fires." .

:Incident rdf:type owl:Class ;
    rdfs:subClassOf [
        rdf:type owl:Class ;
        owl:intersectionOf ( :Incident [
            rdf:type owl:Restriction ;
            owl:onProperty :hasUrgency ;
            owl:someValuesFrom xsd:string
        ] )
    ] .
