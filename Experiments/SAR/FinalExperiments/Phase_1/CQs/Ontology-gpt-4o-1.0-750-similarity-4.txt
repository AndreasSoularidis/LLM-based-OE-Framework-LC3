Y   CQ5. What is the priority of an incident during a forest fire disaster?
    SELECT ?incident ?priority
    WHERE {
        ?priority a sar:Priority .
        ?incident a sar:WildfireIncident ;
            sar:priority ?priority .
    }

Y   CQ6. What incidents during forest fires are the most urgent?
    SELECT ?incident ?urgency
    WHERE {
        ?incident a sar:WildFireIncident ;
            sar:urgency ?urgency .
        ?urgency a sar:Urgent . 
        FILTER(?urgency = "High")
    }

Y   CQ12. What is the status of wildfire forestry works (firebreaks, access to forest roads, etc.)?
    SELECT ?incident ?firebreak_status ?roads_status 
    WHERE {
        ?incident a sar:WildfireIncident ;
            sar:hasFirebreakStatus ?firebreak_status ;
            sar:hasAccessToForestRoads ?roads_status .
        ?firebreak_status a sar:FirebreakStatus .
        ?road_status a sar:AccessToForestRoads .
    }
